>>> MyList = __import__("1-my_list").MyList

my_list = MyList()

# Already sorted list
>>> my_list = MyList()
>>> my_list.append(10)
>>> my_list.append(12)
>>> my_list.append(13)
>>> my_list.append(20)
>>> my_list.print_sorted()
[10, 12, 13, 20]

# Repeated numbers
>>> my_list1 = MyList()
>>> my_list1.append(10)
>>> my_list1.append(12)
>>> my_list1.append(10)
>>> my_list1.append(12)
>>> my_list1.print_sorted()
[10, 10, 12, 12]

# Descending order
>>> lista = MyList()
>>> lista.append(10)
>>> lista.append(9)
>>> lista.append(8)
>>> lista.append(7)
>>> lista.print_sorted()
[7, 8, 9, 10]

# Negative numbers
>>> my_list3 = MyList()
>>> my_list3.append(-10)
>>> my_list3.append(9)
>>> my_list3.append(-8)
>>> my_list3.append(7)
>>> my_list3.print_sorted()
[-10, -8, 7, 9]

#Empty list
>>> emp_list = MyList()
>>> emp_list.print_sorted()
[]
