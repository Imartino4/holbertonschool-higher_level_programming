>>> MyList = __import__("1-my_list").MyList

# Already sorted list
>>> my_list = MyList()
>>> my_list.append(10)
>>> my_list.append(12)
>>> my_list.append(13)
>>> my_list.append(20)
>>> my_list.print_sorted()
[10, 12, 13, 20]

# Repeated numbers
>>> my_list = MyList()
>>> my_list.append(10)
>>> my_list.append(12)
>>> my_list.append(10)
>>> my_list.append(12)
>>> my_list.print_sorted()
[10, 10, 12, 12]

# Descending order
>>> my_list = MyList()
>>> my_list.append(10)
>>> my_list.append(9)
>>> my_list.append(8)
>>> my_list.append(7)
>>> my_list.print_sorted()
[7, 8, 9, 10]
